CREATE TABLE settings (

key_settings INT(10) UNSIGNED AUTO_INCREMENT PRIMARY KEY,

setting_key VARCHAR(100) NOT NULL, setting_value TEXT NOT NULL,

setting_group VARCHAR(50) DEFAULT 'general',

setting_type ENUM('text','number','boolean','url','color','json') DEFAULT 'text', is_active TINYINT(1) DEFAULT 1,

entry_date_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP

);


üÖß Column Breakdown

Column Purpose

setting_key Unique identifier (e.g. homepage_featured_articles_count) setting_value Actual value (e.g. 5)

setting_group Logical grouping (e.g. homepage, article_view, seo) setting_type Helps frontend render correct input type

is_active Toggle for deprecated or draft settings

entry_date_time Audit trail
